```
AI: Ollama (local analysis)
Infrastructure: Docker + Prometheus
```

**CRE Workflow Example:**
```yaml
workflow: infrastructure_monitor
steps:
  - fetch_server_metrics:
      api: "https://monitoring.swarm.dev/metrics"
      method: GET
  - evaluate_thresholds:
      condition: "cpu > 80% OR memory > 90%"
  - trigger_scaling:
      if: condition_met
      action: scale_up
  - notify_team:
      method: smart_contract_event
      data: "Scaling triggered: {{reason}}"
```
